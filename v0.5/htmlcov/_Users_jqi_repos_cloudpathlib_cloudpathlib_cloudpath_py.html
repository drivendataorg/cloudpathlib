611 def _refresh_cache(self, force_overwrite_from_cloud=False): 

612 # nothing to cache if the file does not exist; happens when creating 

613 # new files that will be uploaded 

614 if not self.exists(): 

615 return 

616 

617 if self.is_dir(): 

618 raise ValueError("Only individual files can be cached") 

619 

620 # if not exist or cloud newer 

621 if ( 

622 not self._local.exists() 

623 or (self._local.stat().st_mtime < self.stat().st_mtime) 

624 or force_overwrite_from_cloud 

625 ): 

626 # ensure there is a home for the file 

627 self._local.parent.mkdir(parents=True, exist_ok=True) 

628 self.download_to(self._local) 

629 

630 # force cache time to match cloud times 

631 os.utime(self._local, times=(self.stat().st_mtime, self.stat().st_mtime)) 

632 

633 if self._dirty: 

634 raise OverwriteDirtyFile( 

635 f"Local file ({self._local}) for cloud path ({self}) has been changed by your code, but " 

636 f"is being requested for download from cloud. Either (1) push your changes to the cloud, " 

637 f"(2) remove the local file, or (3) pass `force_overwrite_from_cloud=True` to " 

638 f"overwrite." 

639 ) 

640 

641 # if local newer but not dirty, it was updated 

642 # by a separate process; do not overwrite unless forced to 

643 if self._local.stat().st_mtime > self.stat().st_mtime: 

644 raise OverwriteNewerLocal( 

645 f"Local file ({self._local}) for cloud path ({self}) is newer on disk, but " 

646 f"is being requested for download from cloud. Either (1) push your changes to the cloud, " 

647 f"(2) remove the local file, or (3) pass `force_overwrite_from_cloud=True` to " 

648 f"overwrite."

651 def _upload_local_to_cloud(self, force_overwrite_to_cloud: bool = False): 

652 # We should never try to be syncing entire directories; we should only 

653 # cache and upload individual files. 

654 if self._local.is_dir(): 

655 raise ValueError("Only individual files can be uploaded to the cloud") 

656 

657 # if cloud does not exist or local is newer or we are overwriting, do the upload 

658 if ( 

659 not self.exists() # cloud does not exist 

660 or (self._local.stat().st_mtime > self.stat().st_mtime) 

661 or force_overwrite_to_cloud 

662 ): 

663 self.client._upload_file( 

664 self._local, 

665 self, 

666 ) 

667 

668 # force cache time to match cloud times 

669 os.utime(self._local, times=(self.stat().st_mtime, self.stat().st_mtime)) 

670 

671 # reset dirty and handle now that this is uploaded 

672 self._dirty = False 

673 self._handle = None 

674 

675 return self 

676 

677 # cloud is newer and we are not overwriting 

678 raise OverwriteNewerCloud( 

679 f"Local file ({self._local}) for cloud path ({self}) is newer in the cloud disk, but " 

680 f"is being requested to be uploaded to the cloud. Either (1) redownload changes from the cloud or " 

681 f"(2) pass `force_overwrite_to_cloud=True` to " 

682 f"overwrite." 

683 )

686# The function resolve is not available on Pure paths because it removes relative 

687# paths and symlinks. We _just_ want the relative path resolution for 

688# cloud paths, so the other logic is removed. Also, we can assume that 

689# cloud paths are absolute. 

690# 

691# Based on resolve from pathlib: 

692# https://github.com/python/cpython/blob/3.8/Lib/pathlib.py#L316-L359 

693def _resolve(path: PurePosixPath) -> str: 

694 sep = "/" 

695 

696 # rebuild path from parts 

697 newpath = "" 

698 for name in str(path).split(sep): 

699 if not name or name == ".": 

700 # current dir, nothing to add 

701 continue 

702 if name == "..": 

703 # parent dir, drop right-most part 

704 newpath, _, _ = newpath.rpartition(sep) 

705 continue 

706 newpath = newpath + sep + name 

707 

708 return newpath or sep
